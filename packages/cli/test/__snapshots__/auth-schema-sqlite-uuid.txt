  image: text("image"),
  createdAt: integer("created_at", { mode: "timestamp_ms" })
    .default(sql`(cast(unixepoch('subsecond') * 1000 as integer))`)
    .notNull(),
  updatedAt: integer("updated_at", { mode: "timestamp_ms" })
    .default(sql`(cast(unixepoch('subsecond') * 1000 as integer))`)
    .$onUpdate(() => /* @__PURE__ */ new Date())
    .notNull(),
  username: text("username").unique(),
  displayUsername: text("display_username"),
});

export const custom_userRelations = relations(custom_user, ({ many }) => ({
  custom_sessions: many(custom_session),
  custom_accounts: many(custom_account),
  signInAttempts: many(signInAttempt),
  twoFactorMethods: many(twoFactorMethod),
  trustedDevices: many(trustedDevice),
}));
